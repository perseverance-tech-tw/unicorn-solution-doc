99 inline void initAudioOnce(const std::string &customFile = "")
100 {
101 using namespace detail;
102 if (ready())
103 return;
104
105 if (!customFile.empty())
106 soundFile() = customFile;
107
108 if (soundFile().empty())
109 {
110 if (std::filesystem::exists("/usr/share/sounds/freedesktop/stereo/bell.oga"))
111 soundFile() = "/usr/share/sounds/freedesktop/stereo/bell.oga";
112 else if (std::filesystem::exists("/usr/share/sounds/alsa/Noise.wav"))
113 soundFile() = "/usr/share/sounds/alsa/Noise.wav"; // cadangan
114 }
115
116 std::string ext;
117 if (auto pos = soundFile().rfind('.'); pos != std::string::npos)
118 ext = soundFile().substr(pos + 1);
119
120 if ((ext == "ogg" || ext == "oga") && which("paplay"))
121 {
122 player() = "paplay";
123 cmdTemplate() = "paplay '%FILE%' >/dev/null 2>&1";
124 }
125 else if (which("aplay"))
126 {
127 player() = "aplay";
128 cmdTemplate() = "aplay -q '%FILE%' >/dev/null 2>&1";
129 }
130 else if (which("paplay"))
131 {
132 player() = "paplay";
133 cmdTemplate() = "paplay '%FILE%' >/dev/null 2>&1";
134 }
135 else
136 {
137 player().clear();
138 }
139
140 std::cerr << "[BEEPER] init: file=" << (soundFile().empty() ? "(none)" : soundFile())
141 << " player=" << (player().empty() ? "(BEL)" : player()) << "\n";
142 ready() = true;

189 inline void stop(const std::string &key)
190 {
191 using namespace detail;
192 std::unique_lock<std::mutex> lk(mtx());
193 auto itR = running().find(key);
194 auto itT = threads().find(key);
195 if (itR == running().end())
196 return;
197
198 itR->second.store(false, std::memory_order_release);
199 std::thread th;
200 if (itT != threads().end())
201 {
202 th = std::move(itT->second);
203 threads().erase(itT);
204 }
205 running().erase(itR);
206 lk.unlock();
207 if (th.joinable())
208 th.join();
209 }

260 inline void update(const std::string &key, VRUDetector::BeepPattern pat, const Config &cfg = {})
261 {
262 using namespace detail;
263
264 if (pat != VRUDetector::BeepPattern::BEEP_3_LOOP)
265 {
266 stop(key);
267 }
268
269 switch (pat)
270 {
271 case VRUDetector::BeepPattern::NONE:
272 return;
273
274 case VRUDetector::BeepPattern::BEEP_2_ONCE:
275 {
276 std::thread([cfg]
277 {
278 beep_once();
279 std::this_thread::sleep_for(std::chrono::milliseconds(cfg.betweenBeepsMs));
280 beep_once(); })
281 .detach();
282 return;
283 }
284
285 case VRUDetector::BeepPattern::BEEP_3_LOOP:
286 {
287 std::lock_guard<std::mutex> lk(mtx());
288 if (running().count(key) && running()[key].load(std::memory_order_acquire))
289 return;
290
291 running()[key].store(true, std::memory_order_release);
292 threads()[key] = std::thread([key, cfg]
293 {
294 while (true) {
295 {
296 std::lock_guard<std::mutex> lk(mtx());
297 if (!running()[key].load(std::memory_order_acquire)) break;
298 }
299 beep_once();
300 std::this_thread::sleep_for(std::chrono::milliseconds(cfg.betweenBeepsMs));
301 beep_once();
302 std::this_thread::sleep_for(std::chrono::milliseconds(cfg.betweenBeepsMs));
303 beep_once();
304
305 std::this_thread::sleep_for(std::chrono::milliseconds(cfg.loopPeriodMs));
306 } });
307 return;
308 }
